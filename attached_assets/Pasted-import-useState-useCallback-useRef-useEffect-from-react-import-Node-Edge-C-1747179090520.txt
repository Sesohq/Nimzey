// Enhanced Node Chain Utility
function getNodeChain(
  targetNodeId: string, 
  nodes: Node[], 
  edges: Edge[]
): { nodes: Node[], edges: Edge[] } {
  const resultNodes: Node[] = [];
  const resultEdges: Edge[] = [];
  const visited = new Set<string>();

  // Start with the target node
  const startNode = nodes.find(n => n.id === targetNodeId);
  if (!startNode) return { nodes: [], edges: [] };

  // Recursive function to trace back dependencies
  function traceBackDependencies(nodeId: string) {
    if (visited.has(nodeId)) return;
    visited.add(nodeId);

    // Find the node
    const node = nodes.find(n => n.id === nodeId);
    if (!node) return;

    // Add the node to the result
    resultNodes.unshift(node);

    // Find incoming edges
    const incomingEdges = edges.filter(e => e.target === nodeId);
    
    // Add incoming edges
    resultEdges.unshift(...incomingEdges);

    // Recursively trace back for each source node
    incomingEdges.forEach(edge => {
      traceBackDependencies(edge.source);
    });
  }

  // Start tracing from the target node
  traceBackDependencies(targetNodeId);

  return { nodes: resultNodes, edges: resultEdges };
}

export function useFilterGraph() {
  // State management
  const [nodes, setNodes] = useState<Node[]>([]);
  const [edges, setEdges] = useState<Edge[]>([]);
  const [selectedNodeId, setSelectedNodeId] = useState<string | null>(null);
  const [sourceImage, setSourceImage] = useState<string | null>(null);
  const [processedImage, setProcessedImage] = useState<string | null>(null);
  const [nodePreview, setNodePreview] = useState<string | null>(null);

  // Refs
  const sourceImageRef = useRef<HTMLImageElement | null>(null);
  const canvasRef = useRef<HTMLCanvasElement | null>(null);

  // Generate Node-Specific Preview
  const generateNodePreview = useCallback((targetNode: Node) => {
    if (!sourceImageRef.current) return;

    try {
      // Create canvas for preview
      const previewCanvas = document.createElement('canvas');
      previewCanvas.width = 300;
      previewCanvas.height = 300;

      // Find the chain of nodes leading to this node
      const nodeChain = getNodeChain(targetNode.id, nodes, edges);

      // Process filters up to this node
      const result = applyFilters(
        sourceImageRef.current,
        nodeChain.nodes,
        nodeChain.edges,
        previewCanvas,
        targetNode.id  // Stop processing at this node
      );

      // Update node preview and main preview
      setNodePreview(result);
      
      // Update the specific node's preview data
      setNodes(prevNodes => 
        prevNodes.map(node => 
          node.id === targetNode.id 
            ? { 
                ...node, 
                data: { 
                  ...node.data, 
                  preview: result 
                } 
              } 
            : node
        )
      );
    } catch (error) {
      console.error('Preview generation error:', error);
      setNodePreview(null);
    }
  }, [nodes, edges, sourceImageRef]);

  // Debounced Preview Update
  const debouncedUpdatePreviews = useCallback(
    debounce(() => {
      // Regenerate previews for all nodes
      if (selectedNodeId) {
        const selectedNode = nodes.find(n => n.id === selectedNodeId);
        if (selectedNode) {
          generateNodePreview(selectedNode);
        }
      }
    }, 300),
    [selectedNodeId, nodes, generateNodePreview]
  );

  // Handle Parameter Changes
  const handleParamChange = useCallback((nodeId: string, paramName: string, value: any) => {
    // Update node parameters
    setNodes(prevNodes => 
      prevNodes.map(node => {
        if (node.id === nodeId) {
          const updatedData = {
            ...node.data,
            params: (node.data as FilterNodeData).params.map(param => 
              param.name === paramName ? { ...param, value } : param
            )
          };
          return { ...node, data: updatedData };
        }
        return node;
      })
    );

    // Trigger preview update
    debouncedUpdatePreviews();
  }, [debouncedUpdatePreviews]);

  // Process Full Image
  const processImage = useCallback(() => {
    if (!sourceImageRef.current) return;

    // Create canvas for full image processing
    const processCanvas = document.createElement('canvas');
    processCanvas.width = sourceImageRef.current.width;
    processCanvas.height = sourceImageRef.current.height;

    // Process full filter chain
    const result = applyFilters(
      sourceImageRef.current,
      nodes,
      edges,
      processCanvas
    );

    // Update processed image
    setProcessedImage(result);
  }, [nodes, edges]);

  // Effect to update source image ref
  useEffect(() => {
    if (sourceImage) {
      const img = new Image();
      img.onload = () => {
        sourceImageRef.current = img;
        processImage();
      };
      img.src = sourceImage;
    }
  }, [sourceImage, processImage]);

  // Effect to update previews when nodes or selected node change
  useEffect(() => {
    if (selectedNodeId) {
      const selectedNode = nodes.find(n => n.id === selectedNodeId);
      if (selectedNode) {
        generateNodePreview(selectedNode);
      }
    }
  }, [selectedNodeId, nodes, generateNodePreview]);

  // Expose methods and state
  return {
    nodes,
    edges,
    selectedNodeId,
    sourceImage,
    processedImage,
    nodePreview,
    setNodes,
    setEdges,
    setSelectedNodeId,
    handleParamChange,
    generateNodePreview,
    processImage,
    // Add other necessary methods
  };
}
