2. Create a global utility for directly updating preview images
Create a new file called directUpdatePreview.ts:
typescript// directUpdatePreview.ts

/**
 * Directly updates a node's preview image, bypassing React
 * This function finds the image element by data attribute and forces it to update
 */
export function updateNodePreviewDirectly(nodeId: string, previewUrl: string): boolean {
  // Find the image element by its data attribute
  const imgElement = document.querySelector(`img[data-node-preview-id="${nodeId}"]`) as HTMLImageElement;
  
  if (imgElement) {
    console.log(`Directly updating preview for node ${nodeId}`);
    
    // Add cache busting to ensure the browser loads the new image
    const cacheBuster = `?t=${Date.now()}`;
    const urlWithCacheBuster = previewUrl.includes('?') 
      ? `${previewUrl}&_cache=${Date.now()}` 
      : `${previewUrl}?_cache=${Date.now()}`;
    
    // Set the src attribute directly
    imgElement.src = urlWithCacheBuster;
    
    return true;
  }
  
  console.log(`Could not find preview image for node ${nodeId}`);
  return false;
}

// Make this accessible globally for debugging
(window as any).updateNodePreviewDirectly = updateNodePreviewDirectly;

5. Make WebGLFilterCanvas listen for these events
typescript// In WebGLFilterCanvas.tsx, add:

useEffect(() => {
  const handlePreviewRequest = async (e: CustomEvent) => {
    if (!rendererRef.current) return;
    
    const nodeId = e.detail?.nodeId;
    if (!nodeId) return;
    
    console.log(`Received preview request for node ${nodeId}`);
    
    try {
      const preview = await rendererRef.current.getNodePreview(nodeId, 300);
      if (preview) {
        updateNodePreviewDirectly(nodeId, preview);
      }
    } catch (err) {
      console.error('Error generating node preview:', err);
    }
  };
  
  // Add event listener
  window.addEventListener('request-node-preview' as any, handlePreviewRequest as EventListener);
  
  // Cleanup
  return () => {
    window.removeEventListener('request-node-preview' as any, handlePreviewRequest as EventListener);
  };
}, []);
This approach:

Completely bypasses React Flow and React state management by directly manipulating the DOM
Uses data attributes to find specific preview images
Updates image URLs directly with cache busting to force refreshes
Uses custom events for communication between components
Keeps the existing React state updates for consistency

This is a "brute force" solution, but it should work even if all our previous approaches failed. The key insight is to stop trying to update React state to trigger re-renders and instead directly manipulate the DOM elements that display the previews.
