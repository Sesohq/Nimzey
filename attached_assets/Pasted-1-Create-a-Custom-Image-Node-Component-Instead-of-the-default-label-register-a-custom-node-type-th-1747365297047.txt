1. Create a Custom Image Node Component
Instead of the default label, register a custom node type that renders an <input type="file"> (hidden) and a clickable wrapper:

Edit
// ImageNode.jsx
import React, { useRef } from 'react';
import { Handle, Position } from 'reactflow';

export function ImageNode({ id, data, selected, isConnectable }) {
  const fileInputRef = useRef(null);

  const onClick = (e) => {
    // Prevent React Flow from starting a drag/connection
    e.stopPropagation();
    fileInputRef.current.click();
  };

  const onFileChange = (e) => {
    const file = e.target.files?.[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = () => {
      data.onImageUpload(id, reader.result);  // callback to update node data
    };
    reader.readAsDataURL(file);
  };

  return (
    <div
      onClick={onClick}
      style={{
        padding: 10,
        border: selected ? '2px solid #007ACC' : '1px solid #aaa',
        borderRadius: 4,
        background: '#fff',
        cursor: 'pointer',
        textAlign: 'center',
        width: 150
      }}
    >
      <input
        ref={fileInputRef}
        type="file"
        accept="image/*"
        style={{ display: 'none' }}
        onChange={onFileChange}
      />

      {data.image ? (
        <img src={data.image} alt="texture" style={{ width: '100%', borderRadius: 4 }} />
      ) : (
        <div style={{ color: '#888' }}>Click to upload texture</div>
      )}

      {/* Add your handles as needed */}
      <Handle type="target" position={Position.Top} isConnectable={isConnectable} />
      <Handle type="source" position={Position.Bottom} isConnectable={isConnectable} />
    </div>
  );
}
Key Points
e.stopPropagation() in onClick prevents React Flow’s drag/selection logic from hijacking your click.

A hidden file input is triggered programmatically.

data.onImageUpload is a function you pass in when creating the node—to let your main component update that node’s data.image URL.

2. Register That Node in Your React Flow Instance

// App.jsx
import React, { useState, useCallback } from 'react';
import ReactFlow, { addEdge, Controls } from 'reactflow';
import { ImageNode } from './ImageNode';
import 'reactflow/dist/style.css';

const nodeTypes = { imageNode: ImageNode };

function App() {
  const [nodes, setNodes] = useState([
    {
      id: '1',
      type: 'imageNode',
      position: { x: 100, y: 100 },
      data: { image: null, onImageUpload: handleImageUpload }
    },
    // ...other nodes
  ]);
  const [edges, setEdges] = useState([]);

  // Called by ImageNode when a file is picked
  function handleImageUpload(nodeId, dataUrl) {
    setNodes((nds) =>
      nds.map((n) =>
        n.id === nodeId ? { ...n, data: { ...n.data, image: dataUrl } } : n
      )
    );
  }

  const onConnect = useCallback((params) => setEdges((eds) => addEdge(params, eds)), []);
  
  return (
    <ReactFlow
      nodes={nodes}
      edges={edges}
      onNodesChange={setNodes}
      onEdgesChange={setEdges}
      onConnect={onConnect}
      nodeTypes={nodeTypes}
      fitView
    >
      <Controls />
    </ReactFlow>
  );
}
3. Why This Works
Custom Node
By defining an imageNode type, you control exactly what’s rendered and where clicks go.

Hidden File Input
The <input> stays out of sight but can be opened programmatically.

Event Handling
Stopping propagation on the wrapper ensures React Flow doesn’t intercept the click for dragging or edge-creation.

State Update
Your handleImageUpload writes the selected file into node.data.image, triggering a re-render to show the texture.
