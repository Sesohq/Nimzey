// In filterAlgorithms.ts, modify the applyFilters function to support
// targeting a specific node and use clear caching logic

export function applyFilters(
  sourceImage: HTMLImageElement,
  nodes: Node[],
  edges: Edge[],
  canvas: HTMLCanvasElement,
  targetNodeId?: string,
  clearCache: boolean = false
): string {
  // Initialize canvas
  const ctx = canvas.getContext('2d');
  if (!ctx) return '';
  
  // Clear canvas
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  
  // Handle cache control
  if (clearCache) {
    nodeResultCache.clear();
  }
  
  // Create a map of node processing order - we'll use a modified topological sort
  // that respects the node chain we're trying to visualize
  const processOrder = getNodeProcessingOrder(nodes, edges, targetNodeId);
  
  // First, cache the source image
  const sourceNode = nodes.find(node => node.type === 'imageNode');
  if (sourceNode) {
    // Create a source canvas
    const sourceCanvas = document.createElement('canvas');
    sourceCanvas.width = canvas.width;
    sourceCanvas.height = canvas.height;
    const sourceCtx = sourceCanvas.getContext('2d');
    
    if (sourceCtx) {
      // Draw the source image and store in cache
      sourceCtx.drawImage(sourceImage, 0, 0, sourceCanvas.width, sourceCanvas.height);
      nodeResultCache.set(sourceNode.id, sourceCanvas);
    }
  }
  
  // Process each node in order
  for (const nodeId of processOrder) {
    // Skip if not needed for this preview
    if (targetNodeId && !isNodeInChain(nodeId, targetNodeId, nodes, edges)) {
      continue;
    }
    
    const node = nodes.find(n => n.id === nodeId);
    if (!node || node.type === 'imageNode' || (node.data.enabled === false)) {
      continue;
    }
    
    try {
      if (node.type === 'filterNode') {
        processFilterNode(nodeId, nodes, edges);
      } else if (node.type === 'blendNode') {
        processBlendNode(nodeId, nodes, edges);
      }
      
      // If this is the target node or we're at the end of processing,
      // draw the result to the output canvas
      if (nodeId === targetNodeId || (!targetNodeId && isLeafNode(nodeId, edges))) {
        const result = nodeResultCache.get(nodeId);
        if (result) {
          ctx.drawImage(result, 0, 0);
        }
      }
    } catch (error) {
      console.error(`Error processing node ${nodeId}:`, error);
    }
  }
  
  // Return the data URL
  return canvas.toDataURL();
}

// Helper function to check if a node is in the chain for a target node
function isNodeInChain(nodeId: string, targetNodeId: string, nodes: Node[], edges: Edge[]): boolean {
  if (nodeId === targetNodeId) return true;
  
  // Build a dependency graph (what nodes depend on this node)
  const dependents: Record<string, string[]> = {};
  nodes.forEach(node => {
    dependents[node.id] = [];
  });
  
  edges.forEach(edge => {
    if (dependents[edge.source]) {
      dependents[edge.source].push(edge.target);
    }
  });
  
  // Use BFS to find if targetNodeId can be reached from nodeId
  const visited = new Set<string>();
  const queue = [nodeId];
  
  while (queue.length > 0) {
    const currentId = queue.shift()!;
    visited.add(currentId);
    
    // If we found the target node, return true
    if (currentId === targetNodeId) {
      return true;
    }
    
    // Add all unvisited dependents to the queue
    for (const dependent of dependents[currentId] || []) {
      if (!visited.has(dependent)) {
        queue.push(dependent);
      }
    }
  }
  
  return false;
}

// Check if a node is a leaf node (no outgoing edges)
function isLeafNode(nodeId: string, edges: Edge[]): boolean {
  return !edges.some(edge => edge.source === nodeId);
}

// Get the processing order for nodes
function getNodeProcessingOrder(nodes: Node[], edges: Edge[], targetNodeId?: string): string[] {
  // If we have a target node, start from that node and work backwards
  if (targetNodeId) {
    // Get all nodes in the chain leading to the target
    const chain = new Set<string>();
    const visited = new Set<string>();
    const stack = [targetNodeId];
    
    while (stack.length > 0) {
      const nodeId = stack.pop()!;
      chain.add(nodeId);
      visited.add(nodeId);
      
      // Find all edges pointing to this node
      const incomingEdges = edges.filter(edge => edge.target === nodeId);
      for (const edge of incomingEdges) {
        if (!visited.has(edge.source)) {
          stack.push(edge.source);
        }
      }
    }
    
    // Now do a topological sort on just these nodes
    return topologicalSort(nodes.filter(node => chain.has(node.id)), edges);
  }
  
  // Otherwise, do a normal topological sort
  return topologicalSort(nodes, edges);
}

// Topological sort for nodes
function topologicalSort(nodes: Node[], edges: Edge[]): string[] {
  // Build dependency graph
  const graph: Record<string, string[]> = {};
  nodes.forEach(node => {
    graph[node.id] = [];
  });
  
  edges.forEach(edge => {
    if (graph[edge.target]) {
      graph[edge.target].push(edge.source);
    }
  });
  
  // Topological sort
  const visited: Record<string, boolean> = {};
  const temp: Record<string, boolean> = {};
  const order: string[] = [];
  
  function visit(nodeId: string) {
    if (visited[nodeId]) return;
    if (temp[nodeId]) return; // Cycle detected
    
    temp[nodeId] = true;
    
    for (const depId of graph[nodeId] || []) {
      visit(depId);
    }
    
    temp[nodeId] = false;
    visited[nodeId] = true;
    order.push(nodeId);
  }
  
  // Visit all nodes
  for (const node of nodes) {
    if (!visited[node.id]) {
      visit(node.id);
    }
  }
  
  return order.reverse();
}

// Process a filter node
function processFilterNode(nodeId: string, nodes: Node[], edges: Edge[]): void {
  const node = nodes.find(n => n.id === nodeId);
  if (!node || node.type !== 'filterNode' || !node.data.enabled) return;
  
  // Get input nodes
  const incomingEdges = edges.filter(edge => edge.target === nodeId);
  if (incomingEdges.length === 0 && node.data.filterType !== 'noiseGenerator') {
    console.log(`Node ${nodeId} has no inputs, skipping`);
    return;
  }
  
  // For generator nodes, create a blank canvas
  if (node.data.filterType === 'noiseGenerator') {
    // Create processing canvas
    const processCanvas = document.createElement('canvas');
    processCanvas.width = 300; // Adjust size as needed
    processCanvas.height = 300;
    const processCtx = processCanvas.getContext('2d');
    
    if (processCtx) {
      // Initialize with white background for generator
      processCtx.fillStyle = '#ffffff';
      processCtx.fillRect(0, 0, processCanvas.width, processCanvas.height);
      
      // Apply the generator filter
      applyGeneratorFilter(node, processCanvas, processCtx);
      
      // Cache the result
      nodeResultCache.set(nodeId, processCanvas);
    }
    return;
  }
  
  // Get the first input node that has cached output
  let inputCanvas = null;
  for (const edge of incomingEdges) {
    if (nodeResultCache.has(edge.source)) {
      inputCanvas = nodeResultCache.get(edge.source);
      break;
    }
  }
  
  if (!inputCanvas) {
    console.log(`No input available for node ${nodeId}, skipping`);
    return;
  }
  
  // Create processing canvas
  const processCanvas = document.createElement('canvas');
  processCanvas.width = inputCanvas.width;
  processCanvas.height = inputCanvas.height;
  const processCtx = processCanvas.getContext('2d');
  
  if (processCtx) {
    // Draw the input to our processing canvas
    processCtx.drawImage(inputCanvas, 0, 0);
    
    // Get image data for processing
    const imageData = processCtx.getImageData(
      0, 0, processCanvas.width, processCanvas.height
    );
    
    // Apply the filter
    applyFilter(node.data.filterType, imageData.data, processCanvas.width, processCanvas.height, node.data.params);
    
    // Put the processed data back
    processCtx.putImageData(imageData, 0, 0);
    
    // Cache the result
    nodeResultCache.set(nodeId, processCanvas);
  }
}

// Process a blend node
function processBlendNode(nodeId: string, nodes: Node[], edges: Edge[]): void {
  const node = nodes.find(n => n.id === nodeId);
  if (!node || node.type !== 'blendNode' || !node.data.enabled) return;
  
  // Find foreground and background inputs
  const fgInputEdge = edges.find(edge => 
    edge.target === nodeId && edge.targetHandle === 'foreground'
  );
  
  const bgInputEdge = edges.find(edge => 
    edge.target === nodeId && edge.targetHandle === 'background'
  );
  
  // If we don't have both inputs, we can't blend
  if (!fgInputEdge || !bgInputEdge || 
      !nodeResultCache.has(fgInputEdge.source) || 
      !nodeResultCache.has(bgInputEdge.source)) {
    console.log(`Missing blend inputs for node ${nodeId}, skipping`);
    return;
  }
  
  const fgCanvas = nodeResultCache.get(fgInputEdge.source)!;
  const bgCanvas = nodeResultCache.get(bgInputEdge.source)!;
  
  // Create processing canvas
  const processCanvas = document.createElement('canvas');
  processCanvas.width = Math.max(fgCanvas.width, bgCanvas.width);
  processCanvas.height = Math.max(fgCanvas.height, bgCanvas.height);
  const processCtx = processCanvas.getContext('2d');
  
  if (processCtx) {
    // Draw background first
    processCtx.drawImage(bgCanvas, 0, 0);
    
    // Apply blend mode and opacity
    processCtx.globalCompositeOperation = node.data.blendMode as GlobalCompositeOperation;
    processCtx.globalAlpha = node.data.opacity / 100; // Assuming opacity is 0-100
    
    // Draw foreground with blending
    processCtx.drawImage(fgCanvas, 0, 0);
    
    // Reset context settings
    processCtx.globalCompositeOperation = 'source-over';
    processCtx.globalAlpha = 1.0;
    
    // Cache the result
    nodeResultCache.set(nodeId, processCanvas);
  }
}

// In useFilterGraph.tsx, update the handleParamChange function:

const handleParamChange = useCallback((nodeId: string, paramName: string, value: number | string) => {
  // Skip if it's just a preview update
  if (paramName === 'preview') {
    setNodes(prevNodes => 
      prevNodes.map(node => 
        node.id === nodeId ? { ...node, data: { ...node.data, preview: value } } : node
      )
    );
    return;
  }
  
  // Clear cache for this node
  nodeResultCache.delete(nodeId);
  
  // Update node data
  setNodes(prevNodes => 
    prevNodes.map(node => {
      if (node.id === nodeId) {
        return {
          ...node,
          data: {
            ...node.data,
            params: node.data.params.map((param: any) => 
              param.name === paramName ? { ...param, value } : param
            )
          }
        };
      }
      return node;
    })
  );
  
  // Update previews after a brief delay to allow state to update
  setTimeout(() => {
    // First update this node's preview
    updateNodePreview(nodeId);
    
    // Then process the full image
    processImage();
  }, 10);
}, [nodes, edges]);

// Add this function to update a specific node's preview
const updateNodePreview = useCallback((nodeId: string) => {
  if (!sourceImageRef.current) return;
  
  const node = nodes.find(n => n.id === nodeId);
  if (!node) return;
  
  // Create a small canvas for the preview
  const previewCanvas = document.createElement('canvas');
  previewCanvas.width = 150; // Small size for preview
  previewCanvas.height = 150;
  
  // Process the image chain leading to this node
  const previewUrl = applyFilters(
    sourceImageRef.current,
    nodes,
    edges,
    previewCanvas,
    nodeId // Target this specific node
  );
  
  // Update the node with its preview
  setNodes(prevNodes => 
    prevNodes.map(n => 
      n.id === nodeId ? { ...n, data: { ...n.data, preview: previewUrl } } : n
    )
  );
}, [nodes, edges, sourceImageRef]);

// In your FilterNode.tsx component, add a preview area
// This is a simplified example - adapt to your component structure

function FilterNode({ data, id }: NodeProps<FilterNodeData>) {
  const {
    label,
    filterType,
    params,
    enabled,
    preview,
    onParamChange,
    onToggleEnabled,
    onRemoveNode
  } = data;
  
  return (
    <div className={`filter-node ${!enabled ? 'disabled' : ''}`}>
      <div className="filter-node-header">
        <div className="filter-node-title">{label}</div>
        <div className="filter-node-controls">
          <button 
            className={`toggle-button ${enabled ? 'enabled' : 'disabled'}`}
            onClick={() => onToggleEnabled(id, !enabled)}
          >
            {enabled ? 'ON' : 'OFF'}
          </button>
          <button className="remove-button" onClick={() => onRemoveNode(id)}>
            ×
          </button>
        </div>
      </div>
      
      {/* Preview Area */}
      <div className="filter-node-preview">
        {preview ? (
          <img 
            src={preview} 
            alt={`${label} preview`} 
            className="node-preview-image"
          />
        ) : (
          <div className="node-preview-placeholder">No Preview</div>
        )}
      </div>
      
      {/* Parameters */}
      <div className="filter-node-params">
        {params.map((param) => (
          <div key={param.name} className="param-row">
            <label>{param.label}</label>
            <input
              type="range"
              min={param.min || 0}
              max={param.max || 100}
              step={param.step || 1}
              value={param.value as number}
              onChange={(e) => 
                onParamChange(id, param.name, parseFloat(e.target.value))
              }
              disabled={!enabled}
            />
            <span className="param-value">
              {param.value}{param.unit || ''}
            </span>
          </div>
        ))}
      </div>
      
      {/* Handles */}
      <Handle
        type="target"
        position={Position.Left}
        style={{ background: '#555' }}
        isConnectable={true}
      />
      <Handle
        type="source"
        position={Position.Right}
        style={{ background: '#555' }}
        isConnectable={true}
      />
    </div>
  );
}
