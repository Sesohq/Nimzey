Focused Solution for Node Previews
Here's a step-by-step approach to fix the preview issue:
1. First, modify the FilterNode Component
The FilterNode component needs to properly display the preview once it's generated. Let's make sure it's correctly handling the preview data:
tsx// In your FilterNode.tsx component
function FilterNode({ data, id, selected }: NodeProps<FilterNodeData>) {
  const {
    label,
    filterType,
    params,
    enabled,
    preview, // This is the preview data URL

    onToggleEnabled,
    onRemoveNode
  } = data;

  // Track if we're waiting for a preview to load
  const [isLoadingPreview, setIsLoadingPreview] = useState(!preview);

  useEffect(() => {
    setIsLoadingPreview(!preview);
  }, [preview]);
  
  // Function to retry preview generation
  const handleRetryPreview = () => {
    console.log("Manually retrying preview for node", id);
    // If the node provides an explicit preview update function, use it
    if (data.onTriggerPreviewUpdate) {
      data.onTriggerPreviewUpdate(id);
    }
  };

  return (
    <div className={`filter-node ${!enabled ? 'disabled' : ''} ${selected ? 'selected' : ''}`}>
      <div className="filter-node-header">
        <div className="filter-node-title">{label}</div>
        <div className="filter-node-controls">
          <button 
            className={`toggle-button ${enabled ? 'enabled' : 'disabled'}`}
            onClick={() => onToggleEnabled(id, !enabled)}
          >
            {enabled ? 'ON' : 'OFF'}
          </button>
          <button className="remove-button" onClick={() => onRemoveNode(id)}>
            ×
          </button>
        </div>
      </div>

      <div className="filter-node-preview">
        {preview ? (
          <img 
            src={preview} 
            alt={`${label} preview`} 
            className="node-preview-image"
            onError={() => setIsLoadingPreview(true)} // Handle image load errors
          />
        ) : isLoadingPreview ? (
          <div className="node-preview-loading">
            <div>Preview generating...</div>
            <div className="filter-type">{filterType}</div>
            <button 
              className="retry-button" 
              onClick={handleRetryPreview}
            >
              Click to retry preview
            </button>
          </div>
        ) : (
          <div className="node-preview-placeholder">No Preview</div>
        )}
      </div>
      
      {/* Parameters */}
      <div className="filter-node-params">
        {params.map((param) => (
          <div key={param.name} className="param-row">
            <label>{param.label}</label>
            <input
              type="range"
              min={param.min || 0}
              max={param.max || 100}
              step={param.step || 1}

2. Add a simplified preview generation directly to filterAlgorithms.ts:
typescript// Add this function to filterAlgorithms.ts
export function generateNodePreview(
  node: Node,
  sourceImage: HTMLImageElement,
  allNodes: Node[],
  allEdges: Edge[]
): string | null {
  // Skip if no source image
  if (!sourceImage) return null;
  
  console.log(`Generating direct preview for node ${node.id}`);
  
  try {
    // Create a small canvas for the preview
    const canvas = document.createElement('canvas');
    canvas.width = 150;
    canvas.height = 150;
    const ctx = canvas.getContext('2d');
    if (!ctx) return null;
    
    // Start with the source image
    ctx.drawImage(sourceImage, 0, 0, canvas.width, canvas.height);
    
    // Apply the filter directly if it's a simple one
    if (node.type === 'filterNode') {
      const data = node.data as FilterNodeData;
      
      // Apply a basic preview effect based on filter type
      switch(data.filterType) {
        case 'blur':
          ctx.filter = `blur(${data.params.find(p => p.name === 'radius')?.value || 5}px)`;
          ctx.drawImage(canvas, 0, 0);
          ctx.filter = 'none';
          break;
          
        case 'grayscale':
          ctx.filter = 'grayscale(100%)';
          ctx.drawImage(canvas, 0, 0);
          ctx.filter = 'none';
          break;
          
        case 'invert':
          ctx.filter = 'invert(100%)';
          ctx.drawImage(canvas, 0, 0);
          ctx.filter = 'none';
          break;
          
        // Add other basic filters as needed
          
        default:
          // For more complex filters, we'll just add a text label
          ctx.fillStyle = 'rgba(0,0,0,0.5)';
          ctx.fillRect(0, 0, canvas.width, 20);
          ctx.fillStyle = '#fff';
          ctx.font = '12px sans-serif';
          ctx.fillText(data.filterType, 5, 15);
          break;
      }
    }
    
    return canvas.toDataURL('image/png');
  } catch (error) {
    console.error(`Error generating preview for node ${node.id}:`, error);
    return null;
  }
}
3. Add a direct preview updater to your useFilterGraph hook:
typescript// In useFilterGraph.tsx or useFilterGraph.ts, add this function

// Simple preview generator for nodes
const updateNodePreviews = useCallback(() => {
  if (!sourceImageRef.current) return;
  
  // Track node updates
  const updates: Record<string, string> = {};
  
  // Update previews for all filter nodes
  nodes.forEach(node => {
    if (node.type === 'filterNode' || node.type === 'blendNode') {
      try {
        // Use a simplified direct preview generator
        const preview = generateNodePreview(node, sourceImageRef.current!, nodes, edges);
        
        if (preview) {
          updates[node.id] = preview;
        }
      } catch (error) {
        console.error(`Error updating preview for node ${node.id}:`, error);
      }
    }
  });
  
  // Batch update all nodes at once
  if (Object.keys(updates).length > 0) {
    setNodes(prevNodes => 
      prevNodes.map(node => 
        updates[node.id] 
          ? { ...node, data: { ...node.data, preview: updates[node.id] } }
          : node
      )
    );
  }
}, [nodes, edges, sourceImageRef.current]);

// Call this when adding a node, changing parameters, or after a short delay when mounting
useEffect(() => {
  if (sourceImageRef.current && nodes.length > 0) {
    const timer = setTimeout(() => {
      updateNodePreviews();
    }, 500);
    return () => clearTimeout(timer);
  }
}, [sourceImageRef.current, nodes.length]);

The Key Benefits of This Approach:

Simple Direct Preview: Generates previews directly in the component rather than relying on the complex chain processing
Fallback System: Shows filter type text if full previews can't be generated
Retry Mechanism: Allows users to retry preview generation if it fails
Minimal Changes: Focuses just on the preview functionality without changing the rest of your architecture
Error Resilience: Handles preview generation errors gracefully

This approach should get previews working in your nodes quickly while minimizing changes to your existing codebase. The previews won't be perfect representations of the full filter chain, but they'll give users a quick visual of what each filter does.
