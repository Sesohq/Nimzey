const FilterNode = ({ data, selected, id }: NodeProps<FilterNodeData>) => {

  const [localPreview, setLocalPreview] = useState<string | null>(data.preview || null);
  
  // Effect to listen for preview updates
  useEffect(() => {
    const handlePreviewUpdate = (e: any) => {

        console.log('Node', id, 'received preview update');
        setLocalPreview(e.detail.preview);
      }

    // Add event listener for our custom event

    // Clean up

3. Add a global helper function to request node previews
Add this to your main index.ts or similar global scope file:
typescript// In a global scope file (or add to window in your app initialization)
window.requestNodePreview = (nodeId: string) => {
  console.log('Requesting preview for node:', nodeId);
  const event = new CustomEvent('request-node-preview', {
    detail: { nodeId }
  });
  window.dispatchEvent(event);
};

4. Make WebGLFilterCanvas listen for preview requests
typescript// In WebGLFilterCanvas.tsx
useEffect(() => {
  const handlePreviewRequest = (e: any) => {
    if (e.detail && e.detail.nodeId && rendererRef.current) {
      console.log('WebGLFilterCanvas received preview request for:', e.detail.nodeId);
      
      // Use a timeout to ensure we're not blocking render
      setTimeout(async () => {
        try {
          const preview = await rendererRef.current!.getNodePreview(e.detail.nodeId, 300);
          if (preview) {
            const previewEvent = new CustomEvent('node-preview-updated', { 
              detail: { nodeId: e.detail.nodeId, preview }
            });
            window.dispatchEvent(previewEvent);
          }
        } catch (err) {
          console.error('Error generating preview:', err);
        }
      }, 0);
    }
  };
  
  window.addEventListener('request-node-preview', handlePreviewRequest);
  
  return () => {
    window.removeEventListener('request-node-preview', handlePreviewRequest);
  };
}, []);
This approach has several advantages:

It sidesteps React's rendering system entirely by using custom DOM events
It doesn't depend on React Flow's node management
It creates a direct communication channel between nodes and the WebGL renderer
Preview requests are triggered by parameter changes immediately
