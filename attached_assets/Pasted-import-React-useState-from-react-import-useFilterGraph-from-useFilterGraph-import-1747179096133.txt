import React, { useState } from 'react';
import { useFilterGraph } from './useFilterGraph';
import ReactFlow, { 
  Controls, 
  Background, 
  NodeProps, 
  EdgeProps 
} from 'reactflow';

// Custom Node Component
const FilterNode: React.FC<NodeProps> = ({ data, id }) => {
  const { handleParamChange } = useFilterGraph();

  return (
    <div className="filter-node">
      <div className="node-header">{data.label}</div>
      
      {/* Node Preview */}
      {data.preview && (
        <img 
          src={data.preview} 
          alt="Node Preview" 
          className="node-preview" 
        />
      )}

      {/* Dynamic Parameter Controls */}
      {data.params?.map((param) => (
        <div key={param.name} className="param-control">
          <label>{param.name}</label>
          {param.type === 'range' && (
            <input 
              type="range"
              value={param.value}
              min={param.min}
              max={param.max}
              step={param.step}
              onChange={(e) => 
                handleParamChange(id, param.name, e.target.value)
              }
            />
          )}
          {param.type === 'select' && (
            <select
              value={param.value}
              onChange={(e) => 
                handleParamChange(id, param.name, e.target.value)
              }
            >
              {param.options?.map((option) => (
                <option key={option} value={option}>
                  {option}
                </option>
              ))}
            </select>
          )}
        </div>
      ))}
    </div>
  );
};
